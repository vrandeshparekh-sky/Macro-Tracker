<script>

    // 1. Default Database

    const defaultDatabase = {

        "Cooked White Rice": { cal: 130, pro: 2.7, carb: 28, fat: 0.3 },

        "Whole Wheat Roti": { cal: 297, pro: 10, carb: 46, fat: 8 },

        "Cooked Toor Dal": { cal: 116, pro: 6, carb: 20, fat: 1 },

        "Raw Chicken Breast": { cal: 165, pro: 31, carb: 0, fat: 3.6 },

        "Eggs (Whole)": { cal: 143, pro: 13, carb: 1, fat: 10 },

        "Paneer (Full Fat)": { cal: 296, pro: 14, carb: 2, fat: 25 },

        "Whey Protein": { cal: 380, pro: 80, carb: 10, fat: 3 }

    };


    // State Variables

    let goals = { cal: 2500, pro: 150, carb: 300, fat: 80 };

    let current = { cal: 0, pro: 0, carb: 0, fat: 0 };

    let foodHistory = [];

    let customDatabase = {};


    // 2. Initialization System

    function loadData() {

        const savedCurrent = localStorage.getItem('macroCurrent');

        const savedHistory = localStorage.getItem('macroHistory');

        const savedCustomDb = localStorage.getItem('macroCustomDb');

        

        if (savedCurrent) current = JSON.parse(savedCurrent);

        if (savedHistory) foodHistory = JSON.parse(savedHistory);

        if (savedCustomDb) customDatabase = JSON.parse(savedCustomDb);

        

        populateDropdown();

        updateDashboard();

        renderTable();

    }


    // 3. Populate Dropdown with Default + Custom Foods

    function populateDropdown() {

        const select = document.getElementById('db-food-name');

        select.innerHTML = ''; 

        

        const mergedDb = { ...defaultDatabase, ...customDatabase };

        

        // Sort alphabetically

        Object.keys(mergedDb).sort().forEach(food => {

            let option = document.createElement('option');

            option.value = food;

            option.textContent = food;

            select.appendChild(option);

        });

    }


    // 4. Save Custom Food

    function saveCustomFood() {

        const name = document.getElementById('custom-name').value.trim();

        const cal = parseFloat(document.getElementById('custom-cal').value) || 0;

        const pro = parseFloat(document.getElementById('custom-pro').value) || 0;

        const carb = parseFloat(document.getElementById('custom-carb').value) || 0;

        const fat = parseFloat(document.getElementById('custom-fat').value) || 0;


        if (!name) return alert("Please enter a food name.");


        customDatabase[name] = { cal, pro, carb, fat };

        localStorage.setItem('macroCustomDb', JSON.stringify(customDatabase));

        

        populateDropdown();

        alert(`${name} has been added to your dropdown menu!`);

        

        // Clear inputs

        document.getElementById('custom-name').value = '';

        document.getElementById('custom-cal').value = '';

        document.getElementById('custom-pro').value = '';

        document.getElementById('custom-carb').value = '';

        document.getElementById('custom-fat').value = '';

    }


    // 5. Add Food to Log

    function addFromDB() {

        const name = document.getElementById('db-food-name').value;

        const grams = parseFloat(document.getElementById('db-food-grams').value);

        if (!grams || grams <= 0) return alert("Please enter a valid weight in grams.");


        const mergedDb = { ...defaultDatabase, ...customDatabase };

        const foodItem = mergedDb[name];

        

        const ratio = grams / 100;

        const consumedCal = Math.round(foodItem.cal * ratio);

        const consumedPro = Math.round(foodItem.pro * ratio);

        const consumedCarb = Math.round(foodItem.carb * ratio);

        const consumedFat = Math.round(foodItem.fat * ratio);


        current.cal += consumedCal;

        current.pro += consumedPro;

        current.carb += consumedCarb;

        current.fat += consumedFat;


        foodHistory.push({ name, grams, cal: consumedCal, pro: consumedPro, carb: consumedCarb, fat: consumedFat });


        saveToStorage();

        updateDashboard();

        renderTable();

        document.getElementById('db-food-grams').value = '';

    }


    // 6. Visual Dashboard & Progress Bars

    function updateDashboard() {

        // Update Text

        document.getElementById('tracker-cal').innerText = `${current.cal} / ${goals.cal}`;

        document.getElementById('tracker-pro').innerText = `${current.pro}g / ${goals.pro}g`;

        document.getElementById('tracker-carb').innerText = `${current.carb}g / ${goals.carb}g`;

        document.getElementById('tracker-fat').innerText = `${current.fat}g / ${goals.fat}g`;


        // Calculate Percentages

        const pctCal = Math.min((current.cal / goals.cal) * 100, 100);

        const pctPro = Math.min((current.pro / goals.pro) * 100, 100);

        const pctCarb = Math.min((current.carb / goals.carb) * 100, 100);

        const pctFat = Math.min((current.fat / goals.fat) * 100, 100);


        // Update Bar Widths

        document.getElementById('bar-cal').style.width = `${pctCal}%`;

        document.getElementById('bar-pro').style.width = `${pctPro}%`;

        document.getElementById('bar-carb').style.width = `${pctCarb}%`;

        document.getElementById('bar-fat').style.width = `${pctFat}%`;


        // Overshoot Warning (Turns bar red if you go over 100%)

        document.getElementById('bar-cal').style.backgroundColor = (current.cal > goals.cal) ? '#e74c3c' : '#85c1e9';

        document.getElementById('bar-pro').style.backgroundColor = (current.pro > goals.pro) ? '#e74c3c' : '#f1948a';

        document.getElementById('bar-carb').style.backgroundColor = (current.carb > goals.carb) ? '#e74c3c' : '#82e0aa';

        document.getElementById('bar-fat').style.backgroundColor = (current.fat > goals.fat) ? '#e74c3c' : '#f7dc6f';

    }


    // 7. Render Table

    function renderTable() {

        const tbody = document.getElementById('food-log');

        tbody.innerHTML = ''; 

        foodHistory.forEach(item => {

            tbody.innerHTML += `<tr><td>${item.name}</td><td>${item.grams}g</td><td>${item.cal}</td><td>${item.pro}g</td><td>${item.carb}g</td><td>${item.fat}g</td></tr>`;

        });

    }


    // 8. Save Data

    function saveToStorage() {

        localStorage.setItem('macroCurrent', JSON.stringify(current));

        localStorage.setItem('macroHistory', JSON.stringify(foodHistory));

    }


    // 9. Clear Day

    function clearDay() {

        if (confirm("Reset totals for a new day?")) {

            current = { cal: 0, pro: 0, carb: 0, fat: 0 };

            foodHistory = [];

            saveToStorage();

            updateDashboard();

            renderTable();

        }

    }


    window.onload = loadData;

</script>
